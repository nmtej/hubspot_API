CREATE TABLE crm_account_links (
  id                   UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id            UUID NOT NULL,

CREATE TABLE crm_contact_links (
  id                    UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id             UUID NOT NULL,
  leadlane_contact_id   UUID NOT NULL,
  crm_system            TEXT NOT NULL,
  crm_contact_id        TEXT NOT NULL,
  email                 TEXT,
  status                TEXT NOT NULL,     -- 'active', 'merged', 'ignored'
  created_at            TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at            TIMESTAMP NOT NULL DEFAULT NOW()
);

ALTER TABLE crm_contact_links

CREATE TABLE IF NOT EXISTS public.crm_webhook_events (
    crm_system      text        NOT NULL,
    event_id        text        NOT NULL,
    occurred_at     timestamptz NULL,
    created_at      timestamptz NOT NULL DEFAULT now(),
    processed_at    timestamptz NULL,
    status          text        NOT NULL DEFAULT 'received',
    last_error      text        NULL,
    PRIMARY KEY (crm_system, event_id)
